Sitecore.Publishing.PublishOptions publishOptions =    new Sitecore.Publishing.PublishOptions(sitecoreItem.Database, Sitecore.Data.Database.GetDatabase("web"), Sitecore.Publishing.PublishMode.SingleItem, sitecoreItem.Language, System.DateTime.Now);  							// Create a publisher with the publishoptionsSitecore.Publishing.Publisher publisher = new Sitecore.Publishing.Publisher(publishOptions);// Choose where to publish from	publisher.Options.RootItem = sitecoreItem;// Publish children as well?	publisher.Options.Deep = true;// Do the publish!publisher.Publish();
